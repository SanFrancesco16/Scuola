    nome varchar(12),
    cognome varchar(12),
    data_nascita date,
    CF_PK varchar(16),
    PRIMARY KEY (CF_PK)
);
CREATE TABLE contocorrente(
    id_cc_PK varchar(16),
    
    PRIMARY KEY (id_cc_PK)
);
CREATE TABLE dipendente(
    CF_PK varchar(16),
    ID_filiale_FK varchar(10),
    ID_agenzia_FK varchar(10),
    PRIMARY KEY (CF_PK)

    ID_PK varchar(10),
    indirizzo varchar(20)
);

    ID_PK varchar(10),
